Update `aruba-leren/src/middleware.ts`:

Current middleware only handles next-intl routing. Need to add Supabase session refresh BEFORE next-intl handling.

Pattern from @supabase/ssr docs:
```ts
import { createServerClient } from '@supabase/ssr'
import { NextResponse, type NextRequest } from 'next/server'
import createIntlMiddleware from 'next-intl/middleware'
import { routing } from './i18n/routing'

const intlMiddleware = createIntlMiddleware(routing)

export async function middleware(request: NextRequest) {
  // Step 1: Supabase session refresh
  let supabaseResponse = NextResponse.next({ request })

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return request.cookies.getAll()
        },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value, options }) => request.cookies.set(name, value))
          supabaseResponse = NextResponse.next({ request })
          cookiesToSet.forEach(({ name, value, options }) =>
            supabaseResponse.cookies.set(name, value, options)
          )
        },
      },
    }
  )

  const { data: { user } } = await supabase.auth.getUser()

  // Step 2: next-intl routing
  return intlMiddleware(request)
}

export const config = {
  matcher: ['/', '/(nl|pap|es)/:path*', '/((?!api|_next|_vercel|.*\\..*).*)'],
}
```

Key: Session refresh happens first (sets cookies), then next-intl handles locale routing. This ensures auth.getUser() refreshes the session cookie on every request (AUTH-04: session persistence).

Note: Currently NOT protecting routes (no redirect if !user). Route protection will be added in Plan 3 when we create the dashboard.
